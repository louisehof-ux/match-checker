Option Explicit ' ============================================= ' MATCH CHECKER — Inclusief Dating Event ' ============================================= ' Tabbladen: "Deelnemers" en "Resultaten" ' Kolommen deelnemers: A=Naam, B=Leeftijd, C=Gender, D=Zoekt ' ============================================= Const MAX_AGE_DIFF As Integer = 10 ' Pas aan naar wens Const MIN_MATCHES As Integer = 2 ' Minimum matches per persoon Sub AnalyseerGroep() Dim wsD As Worksheet, wsR As Worksheet Dim lastRow As Long, i As Long, j As Long Dim matchCount As Integer Dim matchNames As String Dim aName As String, aAge As Integer, aGender As String, aZoekt As String Dim bName As String, bAge As Integer, bGender As String, bZoekt As String ' Tabbladen ophalen of aanmaken On Error Resume Next Set wsD = ThisWorkbook.Sheets("Deelnemers") Set wsR = ThisWorkbook.Sheets("Resultaten") On Error GoTo 0 If wsD Is Nothing Then MsgBox "Tabblad 'Deelnemers' niet gevonden. Maak dit tabblad aan.", vbExclamation Exit Sub End If If wsR Is Nothing Then Set wsR = ThisWorkbook.Sheets.Add(After:=wsD) wsR.Name = "Resultaten" End If ' Resultatenblad leegmaken en opmaken wsR.Cells.Clear With wsR ' Titel .Range("A1").Value = "MATCH CHECKER — RESULTATEN" .Range("A1").Font.Bold = True .Range("A1").Font.Size = 14 .Range("A1").Font.Color = RGB(88, 44, 131) ' Kolomkoppen .Range("A3").Value = "Naam" .Range("B3").Value = "Leeftijd" .Range("C3").Value = "Gender" .Range("D3").Value = "Zoekt" .Range("E3").Value = "Aantal matches" .Range("F3").Value = "Status" .Range("G3").Value = "Mogelijke matches" With .Range("A3:G3") .Font.Bold = True .Interior.Color = RGB(230, 210, 255) End With End With lastRow = wsD.Cells(wsD.Rows.Count, "A").End(xlUp).Row If lastRow < 2 Then MsgBox "Geen deelnemers gevonden. Vul eerst het tabblad 'Deelnemers' in.", vbInformation Exit Sub End If Dim outputRow As Integer outputRow = 4 Dim totalMatches As Integer Dim belowMin As Integer Dim noMatch As Integer totalMatches = 0 belowMin = 0 noMatch = 0 ' Loop door alle deelnemers For i = 2 To lastRow aName = Trim(wsD.Cells(i, 1).Value) aAge = 0 If IsNumeric(wsD.Cells(i, 2).Value) Then aAge = CInt(wsD.Cells(i, 2).Value) aGender = Trim(wsD.Cells(i, 3).Value) aZoekt = Trim(wsD.Cells(i, 4).Value) If aName = "" Then GoTo NextParticipant matchCount = 0 matchNames = "" ' Vergelijk met alle andere deelnemers For j = 2 To lastRow If i = j Then GoTo NextMatch bName = Trim(wsD.Cells(j, 1).Value) bAge = 0 If IsNumeric(wsD.Cells(j, 2).Value) Then bAge = CInt(wsD.Cells(j, 2).Value) bGender = Trim(wsD.Cells(j, 3).Value) bZoekt = Trim(wsD.Cells(j, 4).Value) If bName = "" Then GoTo NextMatch ' Leeftijdscheck If aAge > 0 And bAge > 0 Then If Abs(aAge - bAge) > MAX_AGE_DIFF Then GoTo NextMatch End If ' Compatibiliteitscheck (wederzijds) Dim aWantsB As Boolean, bWantsA As Boolean aWantsB = (aZoekt = "Mens (iedereen)") Or (aZoekt = bGender) bWantsA = (bZoekt = "Mens (iedereen)") Or (bZoekt = aGender) If aWantsB And bWantsA Then matchCount = matchCount + 1 If matchNames = "" Then matchNames = bName & " (" & bAge & " jr, " & bGender & ")" Else matchNames = matchNames & " | " & bName & " (" & bAge & " jr, " & bGender & ")" End If End If NextMatch: Next j ' Wegschrijven naar resultatenblad wsR.Cells(outputRow, 1).Value = aName wsR.Cells(outputRow, 2).Value = IIf(aAge > 0, aAge, "") wsR.Cells(outputRow, 3).Value = aGender wsR.Cells(outputRow, 4).Value = aZoekt wsR.Cells(outputRow, 5).Value = matchCount wsR.Cells(outputRow, 7).Value = IIf(matchNames = "", "—", matchNames) totalMatches = totalMatches + matchCount ' Status + kleur If matchCount = 0 Then wsR.Cells(outputRow, 6).Value = "Geen match" wsR.Cells(outputRow, 6).Interior.Color = RGB(255, 200, 200) wsR.Rows(outputRow).Interior.Color = RGB(255, 235, 235) noMatch = noMatch + 1 ElseIf matchCount < MIN_MATCHES Then wsR.Cells(outputRow, 6).Value = "Onder drempel" wsR.Cells(outputRow, 6).Interior.Color = RGB(255, 220, 150) wsR.Rows(outputRow).Interior.Color = RGB(255, 245, 220) belowMin = belowMin + 1 Else wsR.Cells(outputRow, 6).Value = "OK" wsR.Cells(outputRow, 6).Interior.Color = RGB(180, 240, 180) End If outputRow = outputRow + 1 NextParticipant: Next i ' Samenvatting Dim deelnemers As Integer deelnemers = outputRow - 4 wsR.Cells(outputRow + 1, 1).Value = "SAMENVATTING" wsR.Cells(outputRow + 1, 1).Font.Bold = True wsR.Cells(outputRow + 2, 1).Value = "Totaal deelnemers:" wsR.Cells(outputRow + 2, 2).Value = deelnemers wsR.Cells(outputRow + 3, 1).Value = "Gemiddeld aantal matches:" wsR.Cells(outputRow + 3, 2).Value = IIf(deelnemers > 0, Round(totalMatches / deelnemers, 1), 0) wsR.Cells(outputRow + 4, 1).Value = "Onder drempel (< " & MIN_MATCHES & " matches):" wsR.Cells(outputRow + 4, 2).Value = belowMin wsR.Cells(outputRow + 5, 1).Value = "Zonder match:" wsR.Cells(outputRow + 5, 2).Value = noMatch ' Kolombreedte aanpassen wsR.Columns("A:G").AutoFit ' Wervingsadvies genereren Call GenereerWervingsadvies(wsD, wsR, lastRow, outputRow + 7) ' Naar resultaten navigeren wsR.Activate wsR.Range("A1").Select MsgBox "Analyse voltooid! Bekijk het tabblad 'Resultaten'.", vbInformation, "Match Checker" End Sub Sub GenereerWervingsadvies(wsD As Worksheet, wsR As Worksheet, lastRow As Long, startRow As Long) Dim i As Long, j As Long Dim aName As String, aAge As Integer, aGender As String, aZoekt As String Dim matchCount As Integer Dim adviesRow As Integer adviesRow = startRow Dim heeftAdvies As Boolean heeftAdvies = False wsR.Cells(startRow - 1, 1).Value = "WERVINGSADVIES" wsR.Cells(startRow - 1, 1).Font.Bold = True wsR.Cells(startRow - 1, 1).Font.Color = RGB(180, 80, 0) For i = 2 To lastRow aName = Trim(wsD.Cells(i, 1).Value) aAge = 0 If IsNumeric(wsD.Cells(i, 2).Value) Then aAge = CInt(wsD.Cells(i, 2).Value) aGender = Trim(wsD.Cells(i, 3).Value) aZoekt = Trim(wsD.Cells(i, 4).Value) If aName = "" Then GoTo NextAdv matchCount = 0 For j = 2 To lastRow If i = j Then GoTo NextAdvMatch Dim bName As String, bAge As Integer, bGender As String, bZoekt As String bName = Trim(wsD.Cells(j, 1).Value) bAge = 0 If IsNumeric(wsD.Cells(j, 2).Value) Then bAge = CInt(wsD.Cells(j, 2).Value) bGender = Trim(wsD.Cells(j, 3).Value) bZoekt = Trim(wsD.Cells(j, 4).Value) If bName = "" Then GoTo NextAdvMatch If aAge > 0 And bAge > 0 Then If Abs(aAge - bAge) > MAX_AGE_DIFF Then GoTo NextAdvMatch End If If ((aZoekt = "Mens (iedereen)") Or (aZoekt = bGender)) And _ ((bZoekt = "Mens (iedereen)") Or (bZoekt = aGender)) Then matchCount = matchCount + 1 End If NextAdvMatch: Next j If matchCount < MIN_MATCHES Then Dim needed As Integer needed = MIN_MATCHES - matchCount Dim ageMin As Integer, ageMax As Integer Dim ageStr As String If aAge > 0 Then ageMin = Application.WorksheetFunction.Max(18, aAge - MAX_AGE_DIFF) ageMax = aAge + MAX_AGE_DIFF ageStr = ageMin & "–" & ageMax & " jr" Else ageStr = "onbekend (voeg leeftijd toe)" End If Dim zoektGenders As String If aZoekt = "Mens (iedereen)" Then zoektGenders = "Man, Vrouw of Non-binair" Else zoektGenders = aZoekt End If wsR.Cells(adviesRow, 1).Value = "Voor " & aName & " (" & aGender & ", zoekt " & aZoekt & "):" wsR.Cells(adviesRow, 1).Font.Bold = True wsR.Cells(adviesRow, 1).Font.Color = RGB(180, 80, 0) adviesRow = adviesRow + 1 wsR.Cells(adviesRow, 1).Value = " → Werf " & needed & " deelnemer(s): gender " & zoektGenders & _ ", zoekt " & aGender & " of Mens (iedereen), leeftijd " & ageStr wsR.Cells(adviesRow, 1).Font.Color = RGB(100, 60, 0) adviesRow = adviesRow + 2 heeftAdvies = True End If NextAdv: Next i If Not heeftAdvies Then wsR.Cells(startRow, 1).Value = "✓ De groep is in balans — iedereen heeft minimaal " & MIN_MATCHES & " matches." wsR.Cells(startRow, 1).Font.Color = RGB(0, 140, 0) wsR.Cells(startRow, 1).Font.Bold = True End If End Sub Function StatusIsGeldig(status As String) As Boolean ' ============================================= ' BEHEER: Geldige bestelstatussen ' Voeg hier statussen toe of verwijder ze. ' Alleen deelnemers met een van deze statussen ' worden meegenomen in de analyse. ' Schrijf statussen in kleine letters. ' ============================================= Dim geldig() As String geldig = Split("paid", ",") ' Voorbeelden om later toe te voegen: ' geldig = Split("paid,refunded,complimentary", ",") Dim i As Integer For i = 0 To UBound(geldig) If Trim(geldig(i)) = status Then StatusIsGeldig = True Exit Function End If Next i StatusIsGeldig = False End Function Function ParseTicketnaam(ticketnaam As String, ByRef gender As String, ByRef zoekt As String) As Boolean ' Vertaalt Eventree ticketnaam naar gender + zoekt Dim t As String t = LCase(Trim(ticketnaam)) gender = "" zoekt = "" ' Gender bepalen (wie is de deelnemer) If Left(t, 3) = "man" Then gender = "Man" ElseIf Left(t, 5) = "woman" Then gender = "Vrouw" ElseIf Left(t, 10) = "non-binary" Then gender = "Non-binair" Else ParseTicketnaam = False Exit Function End If ' Zoekt bepalen (wat zoekt de deelnemer) If InStr(t, "looking for men") > 0 And InStr(t, "looking for non") = 0 Then zoekt = "Man" ElseIf InStr(t, "looking for women") > 0 Then zoekt = "Vrouw" ElseIf InStr(t, "looking for non-binary") > 0 Then zoekt = "Non-binair" ElseIf InStr(t, "looking for human being") > 0 Then zoekt = "Mens (iedereen)" Else ParseTicketnaam = False Exit Function End If ParseTicketnaam = True End Function Sub ImporteerEventree() ' Importeert een Eventree Excel-export en leest gender/zoekt uit de ticketnaam Dim filePath As String filePath = Application.GetOpenFilename("Excel bestanden (*.xlsx;*.xls), *.xlsx;*.xls", , "Kies je Eventree Excel export") If filePath = "False" Then Exit Sub ' Eventree bestand openen Dim wbEventree As Workbook Application.ScreenUpdating = False Set wbEventree = Workbooks.Open(filePath, ReadOnly:=True) Dim wsE As Worksheet Set wsE = wbEventree.Sheets(1) ' Kolomnummers ophalen op basis van koptekst Dim lastCol As Integer lastCol = wsE.Cells(1, wsE.Columns.Count).End(xlToLeft).Column Dim colNaam As Integer, colTicket As Integer colNaam = 0 colTicket = 0 Dim c As Integer For c = 1 To lastCol Select Case LCase(Trim(wsE.Cells(1, c).Value)) Case "tickethouder naam": colNaam = c Case "ticketnaam": colTicket = c End Select Next c If colNaam = 0 Or colTicket = 0 Then MsgBox "Kolommen 'Tickethouder naam' of 'Ticketnaam' niet gevonden in het bestand.", vbExclamation wbEventree.Close False Application.ScreenUpdating = True Exit Sub End If ' Deelnemersblad ophalen of aanmaken Dim wsDeel As Worksheet On Error Resume Next Set wsDeel = ThisWorkbook.Sheets("Deelnemers") On Error GoTo 0 If wsDeel Is Nothing Then Call MaakDeelnemersBlad Set wsDeel = ThisWorkbook.Sheets("Deelnemers") End If ' Bestaande data wissen Dim lastExisting As Long lastExisting = wsDeel.Cells(wsDeel.Rows.Count, "A").End(xlUp).Row If lastExisting > 1 Then Dim antw As VbMsgBoxResult antw = MsgBox("Er staan al deelnemers in het blad. Wil je ze vervangen?", vbYesNo + vbQuestion) If antw = vbYes Then wsDeel.Range("A2:D" & lastExisting).ClearContents End If ' Rijen inlezen Dim lastRow As Long lastRow = wsE.Cells(wsE.Rows.Count, colNaam).End(xlUp).Row Dim outputRow As Long outputRow = wsDeel.Cells(wsDeel.Rows.Count, "A").End(xlUp).Row + 1 Dim imported As Integer, skipped As Integer imported = 0 skipped = 0 Dim skipNames As String skipNames = "" ' Kolom Bestelstatus ophalen Dim colStatus As Integer colStatus = 0 For c = 1 To lastCol If LCase(Trim(wsE.Cells(1, c).Value)) = "bestelstatus" Then colStatus = c End If Next c Dim i As Long For i = 2 To lastRow Dim naam As String, ticketnaam As String Dim gender As String, zoekt As String naam = Trim(wsE.Cells(i, colNaam).Value) ticketnaam = Trim(wsE.Cells(i, colTicket).Value) ' Sla over als bestelstatus niet in de lijst van geldige statussen staat If colStatus > 0 Then Dim status As String status = LCase(Trim(wsE.Cells(i, colStatus).Value)) If Not StatusIsGeldig(status) Then GoTo VolgendeRij End If If naam = "" Then GoTo VolgendeRij If ParseTicketnaam(ticketnaam, gender, zoekt) Then wsDeel.Cells(outputRow, 1).Value = naam wsDeel.Cells(outputRow, 2).Value = "" ' Leeftijd niet in Eventree export wsDeel.Cells(outputRow, 3).Value = gender wsDeel.Cells(outputRow, 4).Value = zoekt outputRow = outputRow + 1 imported = imported + 1 Else skipped = skipped + 1 skipNames = skipNames & "- " & naam & " (ticket: " & ticketnaam & ")" & vbNewLine End If VolgendeRij: Next i wbEventree.Close False Application.ScreenUpdating = True Dim msg As String msg = imported & " deelnemers geïmporteerd uit Eventree." & vbNewLine If skipped > 0 Then msg = msg & vbNewLine & skipped & " rij(en) overgeslagen (onbekende ticketnaam):" & vbNewLine & skipNames End If msg = msg & vbNewLine & "Voeg eventueel leeftijden toe en voer daarna de analyse uit." MsgBox msg, vbInformation, "Import voltooid" End Sub Sub MaakKnopImport() ' Voegt een importknop toe naast de analyseknop Dim ws As Worksheet On Error Resume Next Set ws = 100) btn.OnAction = "ImporteerEventree" MsgBox "Importknop toegevoegd!", vbInformation End Sub Sub MaakKnop() ' Voegt een knop toe aan het Deelnemers-tabblad om de analyse te starten Dim ws As ws.Shapes("BtnAnalyseer").Fill.ForeColor.RGB = RGB(88, 44, 131) MsgBox "Knop toegevoegd aan het tabblad 'Deelnemers'!", vbInformation End Sub Sub MaakDeelnemersBlad() ' Hulpmacro: maakt een leeg
